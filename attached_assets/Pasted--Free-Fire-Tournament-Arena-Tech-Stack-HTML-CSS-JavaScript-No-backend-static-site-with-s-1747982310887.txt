*
  Free Fire Tournament Arena
  Tech Stack: HTML | CSS | JavaScript
  No backend—static site with simulated payment flow
*/

1. **Initialize Project**
- Create a new Replit project named "FreeFire Tournament".
- Set up file structure:
  - index.html
  - style.css
  - script.js
  - assets/images/
  - assets/qr-resources/
  - assets/fonts/

2. **index.html Boilerplate**
- Add HTML5 doctype and `<head>` linking style.css, jsPDF CDN, and script.js at bottom.
- Build a **responsive navbar** with links: Home, Tournaments, How to Join, Contact.
- Create a **hero section**: full-screen background image, centered title "Free Fire Tournament Arena", subtitle, and a primary "View Tournaments" button.
- Insert `<main id="tournament-list"></main>` for dynamic cards.
- Add a **hidden modal** (`<div id="payment-modal" class="modal">`) containing:
  • Tournament details (name, fee)
  • Player Name input
  • Radio buttons: UPI ID vs Show QR
  • `<img id="qr-image" src="assets/qr-resources/qr.png" hidden>`
  • Buttons: "Pay via UPI", "Download Receipt" (disabled)
  • Close icon
- Footer with social icons and copyright.

3. **style.css**
- Define CSS variables for primary (e.g. #FF5722), dark background, light text.
- Apply a dark theme, using mobile-first responsive design.
- Make navbar sticky, add smooth-scroll behavior.
- Style hero: `height: 100vh`, `background: url(...) center/cover`, flex-centered text.
- Create **tournament cards** grid: `display: grid; grid-template-columns: repeat(auto-fit, minmax(280px,1fr));` with hover scale and shadow.
- Design modal overlay: fixed, centered, semi-transparent backdrop, rounded card.
- Add media queries for widths <768px.

4. **script.js**
```js
// Sample tournament data
const tournaments = [
  { id: 1, name: "Weekend Slayer", date: "2025-06-01", entryFee: 50, prize: 500, description: "Solo mode, 50 participants." },
  { id: 2, name: "Clash Royale", date: "2025-06-08", entryFee: 100, prize: 1000, description: "Team deathmatch." },
  // add more
];

function renderTournaments() {
  const container = document.getElementById('tournament-list');
  container.innerHTML = '';
  tournaments.forEach(t => {
    const card = document.createElement('div');
    card.className = 'tournament-card';
    card.innerHTML = `
      <h3>${t.name}</h3>
      <p>${t.description}</p>
      <p>Date: ${t.date}</p>
      <p>Entry Fee: ₹${t.entryFee}</p>
      <p>Prize: ₹${t.prize}</p>
      <button class="join-btn" data-id="${t.id}">Join</button>
      <div class="countdown" data-date="${t.date}"></div>
    `;
    container.appendChild(card);
  });
}

document.addEventListener('click', async e => {
  if (e.target.classList.contains('join-btn')) {
    const id = +e.target.dataset.id;
    const tour = tournaments.find(t => t.id === id);
    openPaymentModal(tour);
  }
});

function openPaymentModal(tour) {
  const modal = document.getElementById('payment-modal');
  // populate fields: name, fee, hide/download buttons
  document.getElementById('modal-title').innerText = `Join ${tour.name}`;
  document.getElementById('modal-fee').innerText = `Fee: ₹${tour.entryFee}`;
  document.getElementById('download-btn').disabled = true;
  modal.classList.add('open');
}

// Radio toggle for QR
document.getElementById('upi-radio').addEventListener('change', () => {
  document.getElementById('qr-image').hidden = true;
});
document.getElementById('qr-radio').addEventListener('change', () => {
  document.getElementById('qr-image').hidden = false;
});

// Payment simulation
document.getElementById('pay-btn').addEventListener('click', () => {
  const downloadBtn = document.getElementById('download-btn');
  setTimeout(() => {
    downloadBtn.disabled = false;
    showToast('Payment successful!');
  }, 2000);
});

// jsPDF receipt
import jsPDF from 'https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js';

document.getElementById('download-btn').addEventListener('click', () => {
  const { jsPDF: PDF } = window.jspdf;
  const doc = new PDF();
  const receiptID = Math.floor(Math.random()*1000000);
  doc.text("Payment Receipt", 20, 20);
  doc.text(`Receipt ID: ${receiptID}`, 20, 40);
  doc.text(`Player: ${document.getElementById('player-name').value}`, 20, 60);
  doc.text(`Tournament: ${document.getElementById('modal-title').innerText}`, 20, 80);
  doc.text(`Fee: ${document.getElementById('modal-fee').innerText}`, 20, 100);
  doc.text(`Date: ${new Date().toLocaleString()}`, 20, 120);
  doc.save(`receipt_${receiptID}.pdf`);
});
